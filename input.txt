void/ bubbleSort[int/[] arr, int/n] {
    int/ i = 0;
    int/ j = 0;

    bool/ swapped;

    do/ {
        swapped = false;
        do/ {
            if/[arr[j] gth/ arr[j add/ 1]]{
                swap[arr[j], arr[j add/ 1]];
                swapped = true;
            }
            j = j add/ 1;
        } while/[j lth/ n sub/ (i sub/ 1)]
        i = i add/ 1;
    } while/ [i lth/ (n sub/ 1)]

    if/[swapped eqt/ false]{
        break/;
    }
}
